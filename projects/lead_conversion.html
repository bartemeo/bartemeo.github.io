Logistic regression with 5-fold cross-validation in R using caret:

# Load libraries
library(caret)
library(tidyverse)

# Train/test split
set.seed(42)
index <- createDataPartition(df$converted, p = 0.8, list = FALSE)
train <- df[index, ]
test  <- df[-index, ]

# Logistic regression with 5-fold CV
ctrl  <- trainControl(method = "cv", number = 5)
model <- train(converted ~ ., data = train,
               method  = "glm",
               family  = "binomial",
               trControl = ctrl)

# Feature importance
varImp(model)
